import matplotlib.pyplot as plt
import numpy as np

models = ['Logistic\nRegression', 'Decision\nTree', 'Random\nForest', 'XGBoost']
train_accuracy = [88.10, 100.00, 100.00, 95.20]  # replace with your actual train scores
test_accuracy  = [87.33,  90.00,  89.67, 87.67]

x = np.arange(len(models))
width = 0.35

fig, ax = plt.subplots(figsize=(11, 6))

bars_train = ax.bar(x - width/2, train_accuracy, width, label='Train', color='#85b7eb', edgecolor='white', linewidth=1.2)
bars_test  = ax.bar(x + width/2, test_accuracy,  width, label='Test',  color='#185fa5', edgecolor='white', linewidth=1.2)

# Value labels
for bar in bars_train:
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 0.2,
            f'{bar.get_height():.1f}%', ha='center', va='bottom', fontsize=9, color='#555555')
for bar in bars_test:
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 0.2,
            f'{bar.get_height():.1f}%', ha='center', va='bottom', fontsize=9, color='#555555')

ax.set_ylim(80, 105)
ax.set_xticks(x)
ax.set_xticklabels(models, fontsize=11)
ax.set_ylabel('Accuracy (%)', fontsize=12, color='#555555')
ax.set_title('Train vs Test Accuracy — Model Comparison', fontsize=14, fontweight='bold', pad=20)
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda v, _: f'{v:.0f}%'))
ax.legend(fontsize=11)
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.spines['left'].set_color('#dddddd')
ax.spines['bottom'].set_color('#dddddd')
ax.yaxis.grid(True, linestyle='--', alpha=0.5, color='#cccccc')
ax.set_axisbelow(True)

plt.tight_layout()
plt.savefig('train_test_comparison.png', dpi=150, bbox_inches='tight')
plt.show()
